132  int size () { return numElements; }
133 
138  bool isFull () { return numElements == maxSize; }
139 
144  bool empty () { return (numElements == 0); }
145 
151  bool push (Telement* item) {
152  bool wasFull = isFull ();
153  DEBUG_DBG ("Add element. Buffer was %s\n", wasFull ? "full" : "not full");
154  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
155  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
156  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
157  writeIndex++;
158  if (writeIndex >= maxSize) {
159  writeIndex %= maxSize;
160  }
161  if (wasFull) { // old value is no longer valid
162  readIndex++;
163  if (readIndex >= maxSize) {
164  readIndex %= maxSize;
165  }
166  } else {
167  numElements++;
168  }
169  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
170  return !wasFull;
171  }
172 
177  bool pop () {
178  bool wasEmpty = empty ();
179  DEBUG_DBG ("Remove element. Buffer was %s\n", wasEmpty ? "empty" : "not empty");
180  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
181  if (!wasEmpty) {
182  readIndex++;
183  if (readIndex >= maxSize) {
184  readIndex %= maxSize;
185  }
186  numElements--;
187  }
188  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
189  return !wasEmpty;
190  }
191 
196  Telement* front () {
197  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
198  if (!empty ()) {
199  return &(buffer[readIndex]);
200  } else {
201  return NULL;
202  }
203  }
204 };
