123  int size () { return numElements; }
124 
129  bool isFull () { return numElements == maxSize; }
130 
135  bool empty () { return (numElements == 0); }
136 
142  bool push (Telement *item) {
143  bool wasFull = isFull ();
144  DEBUG_DBG ("Add element. Buffer was %s\n", wasFull ? "full" : "not full");
145  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
146  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
147  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
148  writeIndex++;
149  if (writeIndex >= maxSize) {
150  writeIndex %= maxSize;
151  }
152  if (wasFull) { // old value is no longer valid
153  readIndex++;
154  if (readIndex >= maxSize) {
155  readIndex %= maxSize;
156  }
157  } else {
158  numElements++;
159  }
160  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
161  return !wasFull;
162  }
163 
168  bool pop () {
169  bool wasEmpty = empty ();
170  DEBUG_DBG ("Remove element. Buffer was %s\n", wasEmpty ? "empty" : "not empty");
171  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
172  if (!wasEmpty) {
173  readIndex++;
174  if (readIndex >= maxSize) {
175  readIndex %= maxSize;
176  }
177  numElements--;
178  }
179  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
180  return !wasEmpty;
181  }
182 
187  Telement* front () {
188  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
189  if (!empty ()) {
190  return &(buffer[readIndex]);
191  } else {
192  return NULL;
193  }
194  }
195 };
