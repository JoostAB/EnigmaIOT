127  bool isFull () { return numElements == maxSize; }
128 
133  bool empty () { return (numElements == 0); }

140  bool push (Telement *item) {
141  bool wasFull = isFull ();
142  DEBUG_DBG ("Add element. Buffer was %s\n", wasFull ? "full" : "not full");
143  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
144  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
145  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
146  writeIndex++;
147  if (writeIndex >= maxSize) {
148  writeIndex %= maxSize;
149  }
150  if (wasFull) { // old value is no longer valid
151  readIndex++;
152  if (readIndex >= maxSize) {
153  readIndex %= maxSize;
154  }
155  } else {
156  numElements++;
157  }
158  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
159  return !wasFull;
160  }
161 
166  bool pop () {
167  bool wasEmpty = empty ();
168  DEBUG_DBG ("Remove element. Buffer was %s\n", wasEmpty ? "empty" : "not empty");
169  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
170  if (!wasEmpty) {
171  readIndex++;
172  if (readIndex >= maxSize) {
173  readIndex %= maxSize;
174  }
175  numElements--;
176  }
177  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
178  return !wasEmpty;
179  }
180 
185  Telement* front () {
186  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
187  if (!empty ()) {
188  return &(buffer[readIndex]);
189  } else {
190  return NULL;
191  }
192  }
193 };
