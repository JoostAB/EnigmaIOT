138  int size () { return numElements; }
139 
144  bool isFull () { return numElements == maxSize; }
145 
150  bool empty () { return (numElements == 0); }
151 
157  bool push (Telement* item) {
158  bool wasFull = isFull ();
159  DEBUG_DBG ("Add element. Buffer was %s\n", wasFull ? "full" : "not full");
160  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
161  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
162  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
163  writeIndex++;
164  if (writeIndex >= maxSize) {
165  writeIndex %= maxSize;
166  }
167  if (wasFull) { // old value is no longer valid
168  readIndex++;
169  if (readIndex >= maxSize) {
170  readIndex %= maxSize;
171  }
172  } else {
173  numElements++;
174  }
175  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
176  return !wasFull;
177  }
178 
183  bool pop () {
184  bool wasEmpty = empty ();
185  DEBUG_DBG ("Remove element. Buffer was %s\n", wasEmpty ? "empty" : "not empty");
186  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
187  if (!wasEmpty) {
188  readIndex++;
189  if (readIndex >= maxSize) {
190  readIndex %= maxSize;
191  }
192  numElements--;
193  }
194  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
195  return !wasEmpty;
196  }
197 
202  Telement* front () {
203  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
204  if (!empty ()) {
205  return &(buffer[readIndex]);
206  } else {
207  return NULL;
208  }
209  }
