136  int size () { return numElements; }
137 
142  bool isFull () { return numElements == maxSize; }
143 
148  bool empty () { return (numElements == 0); }
149 
155  bool push (Telement* item) {
156  bool wasFull = isFull ();
157  DEBUG_DBG ("Add element. Buffer was %s\n", wasFull ? "full" : "not full");
158  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
159  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
160  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
161  writeIndex++;
162  if (writeIndex >= maxSize) {
163  writeIndex %= maxSize;
164  }
165  if (wasFull) { // old value is no longer valid
166  readIndex++;
167  if (readIndex >= maxSize) {
168  readIndex %= maxSize;
169  }
170  } else {
171  numElements++;
172  }
173  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
174  return !wasFull;
175  }
176 
181  bool pop () {
182  bool wasEmpty = empty ();
183  DEBUG_DBG ("Remove element. Buffer was %s\n", wasEmpty ? "empty" : "not empty");
184  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
185  if (!wasEmpty) {
186  readIndex++;
187  if (readIndex >= maxSize) {
188  readIndex %= maxSize;
189  }
190  numElements--;
191  }
192  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
193  return !wasEmpty;
194  }
195 
200  Telement* front () {
201  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d\n", readIndex, writeIndex, numElements);
202  if (!empty ()) {
203  return &(buffer[readIndex]);
204  } else {
205  return NULL;
206  }
207  }
