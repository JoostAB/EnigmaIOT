165  bool push (Telement* item) {
166  bool wasFull = isFull ();
167  DEBUG_DBG ("Add element. Buffer was %s", wasFull ? "full" : "not full");
168  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d", readIndex, writeIndex, numElements);
169  memcpy (&(buffer[writeIndex]), item, sizeof (Telement));
170  //Serial.printf ("Copied: %d bytes\n", sizeof (Telement));
171  writeIndex++;
172  if (writeIndex >= maxSize) {
173  writeIndex %= maxSize;
174  }
175  if (wasFull) { // old value is no longer valid
176  readIndex++;
177  if (readIndex >= maxSize) {
178  readIndex %= maxSize;
179  }
180  } else {
181  numElements++;
182  }
183  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d", readIndex, writeIndex, numElements);
184  return !wasFull;
185  }
186 
191  bool pop () {
192  bool wasEmpty = empty ();
193  DEBUG_DBG ("Remove element. Buffer was %s", wasEmpty ? "empty" : "not empty");
194  DEBUG_DBG ("Before -- > ReadIdx: %d. WriteIdx: %d. Size: %d", readIndex, writeIndex, numElements);
195  if (!wasEmpty) {
196  readIndex++;
197  if (readIndex >= maxSize) {
198  readIndex %= maxSize;
199  }
200  numElements--;
201  }
202  DEBUG_DBG ("After -- > ReadIdx: %d. WriteIdx: %d. Size: %d", readIndex, writeIndex, numElements);
203  return !wasEmpty;
204  }
205 
210  Telement* front () {
211  DEBUG_DBG ("Read element. ReadIdx: %d. WriteIdx: %d. Size: %d", readIndex, writeIndex, numElements);
212  if (!empty ()) {
213  return &(buffer[readIndex]);
214  } else {
215  return NULL;
216  }
217  }
