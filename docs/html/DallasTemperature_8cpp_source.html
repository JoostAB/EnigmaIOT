158 bool DallasTemperature::readScratchPad(const uint8_t* deviceAddress,
159  uint8_t* scratchPad) {
160 
161  // send the reset command and fail fast
162  int b = _wire->reset();
163  if (b == 0)
164  return false;
165 
166  _wire->select(deviceAddress);
167  _wire->write(READSCRATCH);
168 
169  // Read all registers in a simple loop
170  // byte 0: temperature LSB
171  // byte 1: temperature MSB
172  // byte 2: high alarm temp
173  // byte 3: low alarm temp
174  // byte 4: DS18S20: store for crc
175  // DS18B20 & DS1822: configuration register
176  // byte 5: internal use & crc
177  // byte 6: DS18S20: COUNT_REMAIN
178  // DS18B20 & DS1822: store for crc
179  // byte 7: DS18S20: COUNT_PER_C
180  // DS18B20 & DS1822: store for crc
181  // byte 8: SCRATCHPAD_CRC
182  for (uint8_t i = 0; i < 9; i++) {
183  scratchPad[i] = _wire->read();
184  }
185 
186  b = _wire->reset();
187  return (b == 1);
188 }

190 void DallasTemperature::writeScratchPad(const uint8_t* deviceAddress,
191  const uint8_t* scratchPad) {
192 
193  _wire->reset();
194  _wire->select(deviceAddress);
195  _wire->write(WRITESCRATCH);
196  _wire->write(scratchPad[HIGH_ALARM_TEMP]); // high alarm temp
197  _wire->write(scratchPad[LOW_ALARM_TEMP]); // low alarm temp
198 
199  // DS1820 and DS18S20 have no configuration register
200  if (deviceAddress[0] != DS18S20MODEL)
201  _wire->write(scratchPad[CONFIGURATION]);
202 
203  _wire->reset();
204 
205  // save the newly written values to eeprom
206  _wire->select(deviceAddress);
207  _wire->write(COPYSCRATCH, parasite);
208  delay(20); // <--- added 20ms delay to allow 10ms long EEPROM write operation (as specified by datasheet)
209 
210  if (parasite)
211  delay(10); // 10ms delay
212  _wire->reset();
213 
214 }

242 // set resolution of a device to 9, 10, 11, or 12 bits
243 // if new resolution is out of range, 9 bits is used.
244 bool DallasTemperature::setResolution(const uint8_t* deviceAddress,
245  uint8_t newResolution, bool skipGlobalBitResolutionCalculation) {
246 
247  // ensure same behavior as setResolution(uint8_t newResolution)
248  newResolution = constrain(newResolution, 9, 12);
249 
250  // return when stored value == new value
251  if (getResolution(deviceAddress) == newResolution)
252  return true;
253 
254  ScratchPad scratchPad;
255  if (isConnected(deviceAddress, scratchPad)) {
256 
257  // DS1820 and DS18S20 have no resolution configuration register
258  if (deviceAddress[0] != DS18S20MODEL) {
259 
260  switch (newResolution) {
261  case 12:
262  scratchPad[CONFIGURATION] = TEMP_12_BIT;
263  break;
264  case 11:
265  scratchPad[CONFIGURATION] = TEMP_11_BIT;
266  break;
267  case 10:
268  scratchPad[CONFIGURATION] = TEMP_10_BIT;
269  break;
270  case 9:
271  default:
272  scratchPad[CONFIGURATION] = TEMP_9_BIT;
273  break;
274  }
275  writeScratchPad(deviceAddress, scratchPad);
276 
277  // without calculation we can always set it to max
278  bitResolution = max(bitResolution, newResolution);
279 
280  if (!skipGlobalBitResolutionCalculation
281  && (bitResolution > newResolution)) {
282  bitResolution = newResolution;
283  DeviceAddress deviceAddr;
284  for (int i = 0; i < devices; i++) {
285  getAddress(deviceAddr, i);
286  bitResolution = max(bitResolution,
287  getResolution(deviceAddr));
288  }
289  }
290  }
291  return true; // new value set
292  }
293 
294  return false;
295 
296 }

466 // reads scratchpad and returns fixed-point temperature, scaling factor 2^-7
467 int16_t DallasTemperature::calculateTemperature(const uint8_t* deviceAddress,
468  uint8_t* scratchPad) {
469 
470  int16_t fpTemperature = (((int16_t) scratchPad[TEMP_MSB]) << 11)
471  | (((int16_t) scratchPad[TEMP_LSB]) << 3);
472 
473  /*
474  DS1820 and DS18S20 have a 9-bit temperature register.
475 
476  Resolutions greater than 9-bit can be calculated using the data from
477  the temperature, and COUNT REMAIN and COUNT PER °C registers in the
478  scratchpad. The resolution of the calculation depends on the model.
479 
480  While the COUNT PER °C register is hard-wired to 16 (10h) in a
481  DS18S20, it changes with temperature in DS1820.
482 
483  After reading the scratchpad, the TEMP_READ value is obtained by
484  truncating the 0.5°C bit (bit 0) from the temperature data. The
485  extended resolution temperature can then be calculated using the
486  following equation:
487 
488  COUNT_PER_C - COUNT_REMAIN
489  TEMPERATURE = TEMP_READ - 0.25 + --------------------------
490  COUNT_PER_C
491 
492  Hagai Shatz simplified this to integer arithmetic for a 12 bits
493  value for a DS18S20, and James Cameron added legacy DS1820 support.
494 
495  See - http://myarduinotoy.blogspot.co.uk/2013/02/12bit-result-from-ds18s20.html
496  */
497 
498  if (deviceAddress[0] == DS18S20MODEL) {
499  fpTemperature = ((fpTemperature & 0xfff0) << 3) - 16
500  + (((scratchPad[COUNT_PER_C] - scratchPad[COUNT_REMAIN]) << 7)
501  / scratchPad[COUNT_PER_C]);
502  }
503 
504  return fpTemperature;
505 }

621  ALARMS:
622 
623  TH and TL Register Format
624 
625  BIT 7 BIT 6 BIT 5 BIT 4 BIT 3 BIT 2 BIT 1 BIT 0
626  S 2^6 2^5 2^4 2^3 2^2 2^1 2^0
627 
628  Only bits 11 through 4 of the temperature register are used

633  updated after every temperature measurement; therefore, if the
634  alarm condition goes away, the flag will be turned off after

725 // Perform an alarm search. If this function returns a '1' then it has
726 // enumerated the next device and you may retrieve the ROM from the
727 // OneWire::address variable. If there are no devices, no further
728 // devices, or something horrible happens in the middle of the
729 // enumeration then a 0 is returned. If a new device is found then
730 // its address is copied to newAddr. Use
731 // DallasTemperature::resetAlarmSearch() to start over.
732 bool DallasTemperature::alarmSearch(uint8_t* newAddr) {
733 
734  uint8_t i;
735  int8_t lastJunction = -1;
736  uint8_t done = 1;
737 
738  if (alarmSearchExhausted)
739  return false;
740  if (!_wire->reset())
741  return false;
742 
743  // send the alarm search command
744  _wire->write(0xEC, 0);
745 
746  for (i = 0; i < 64; i++) {
747 
748  uint8_t a = _wire->read_bit();
749  uint8_t nota = _wire->read_bit();
750  uint8_t ibyte = i / 8;
751  uint8_t ibit = 1 << (i & 7);
752 
753  // I don't think this should happen, this means nothing responded, but maybe if
754  // something vanishes during the search it will come up.
755  if (a && nota)
756  return false;
757 
758  if (!a && !nota) {
759  if (i == alarmSearchJunction) {
760  // this is our time to decide differently, we went zero last time, go one.
761  a = 1;
762  alarmSearchJunction = lastJunction;
763  } else if (i < alarmSearchJunction) {
764 
765  // take whatever we took last time, look in address
766  if (alarmSearchAddress[ibyte] & ibit) {
767  a = 1;
768  } else {
769  // Only 0s count as pending junctions, we've already exhausted the 0 side of 1s
770  a = 0;
771  done = 0;
772  lastJunction = i;
773  }
774  } else {
775  // we are blazing new tree, take the 0
776  a = 0;
777  alarmSearchJunction = i;
778  done = 0;
779  }
780  // OneWire search fix
781  // See: http://www.arduino.cc/cgi-bin/yabb2/YaBB.pl?num=1238032295
782  }
783 
784  if (a)
785  alarmSearchAddress[ibyte] |= ibit;
786  else
787  alarmSearchAddress[ibyte] &= ~ibit;
788 
789  _wire->write_bit(a);
790  }
791 
792  if (done)
793  alarmSearchExhausted = 1;
794  for (i = 0; i < 8; i++)
795  newAddr[i] = alarmSearchAddress[i];
796  return true;
797 
798 }
