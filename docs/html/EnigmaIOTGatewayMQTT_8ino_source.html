191 void processRxData (uint8_t* mac, uint8_t* buffer, uint8_t length, uint16_t lostMessages, bool control, gatewayPayloadEncoding_t payload_type) {
192  uint8_t* addr = mac;
193  char* payload;
194  size_t pld_size;
195  const int PAYLOAD_SIZE = 1024; // Max MQTT payload in PubSubClient library normal operation.
196 
197  payload = (char*)malloc (PAYLOAD_SIZE);
198 
199  char mac_str[18];
200  mac2str (addr, mac_str);
201  if (control) {
202  processRxControlData (mac_str, buffer, length);
203  return;
204  }
205  //char* netName = EnigmaIOTGateway.getNetworkName ();
206  if (payload_type == CAYENNELPP) {
207  DEBUG_INFO ("CayenneLPP message");
208  const int capacity = JSON_ARRAY_SIZE (25) + 25 * JSON_OBJECT_SIZE (4);
209  DynamicJsonDocument jsonBuffer (capacity);
210  JsonArray root = jsonBuffer.createNestedArray ();
211  CayenneLPP cayennelpp (MAX_DATA_PAYLOAD_SIZE);
212 
213  cayennelpp.decode ((uint8_t*)buffer, length, root);
214  uint8_t error = cayennelpp.getError ();
215  if (error != LPP_ERROR_OK) {
216  DEBUG_ERROR ("Error decoding CayenneLPP data: %d", error);
217  return;
218  }
219  pld_size = serializeJson (root, payload, PAYLOAD_SIZE);
220  } else if (payload_type == MSG_PACK) {
221  DEBUG_INFO ("MsgPack message");
222  const int capacity = JSON_ARRAY_SIZE (25) + 25 * JSON_OBJECT_SIZE (4);
223  DynamicJsonDocument jsonBuffer (capacity);
224  DeserializationError error = deserializeMsgPack (jsonBuffer, buffer, length);
225  if (error != DeserializationError::Ok) {
226  DEBUG_ERROR ("Error decoding MSG Pack data: %s", error.c_str ());
227  return;
228  }
229  pld_size = serializeJson (jsonBuffer, payload, PAYLOAD_SIZE);
230  } else if (payload_type == RAW) {
231  DEBUG_INFO ("RAW message");
232  if (length <= PAYLOAD_SIZE) {
233  memcpy (payload, buffer, length);
234  pld_size = length;
235  } else { // This will not happen but may lead to errors in case of using another physical transport
236  memcpy (payload, buffer, PAYLOAD_SIZE);
237  pld_size = PAYLOAD_SIZE;
238  }
239  }
240 
241  GwOutput.outputDataSend (mac_str, payload, pld_size);
242  DEBUG_INFO ("Published data message from %s, length %d: %s, Encoding 0x%02X", mac_str, pld_size, payload, payload_type);
243  if (lostMessages > 0) {
244  pld_size = snprintf (payload, PAYLOAD_SIZE, "%u", lostMessages);
245  GwOutput.outputDataSend (mac_str, payload, pld_size, GwOutput_data_type::lostmessages);
246  DEBUG_INFO ("Published MQTT from %s: %s", mac_str, payload);
247  }
248  pld_size = snprintf (payload, PAYLOAD_SIZE, "{\"per\":%e,\"lostmessages\":%u,\"totalmessages\":%u,\"packetshour\":%.2f}",
249  EnigmaIOTGateway.getPER ((uint8_t*)mac),
250  EnigmaIOTGateway.getErrorPackets ((uint8_t*)mac),
251  EnigmaIOTGateway.getTotalPackets ((uint8_t*)mac),
252  EnigmaIOTGateway.getPacketsHour ((uint8_t*)mac));
253  GwOutput.outputDataSend (mac_str, payload, pld_size, GwOutput_data_type::status);
254  DEBUG_INFO ("Published MQTT from %s: %s", mac_str, payload);
255  free (payload);
256 }

258 void onDownlinkData (uint8_t* address, control_message_type_t msgType, char* data, unsigned int len){
259  uint8_t *buffer;
260  unsigned int bufferLen = len;
261  gatewayPayloadEncoding_t encoding = ENIGMAIOT;
262 
263 
264  DEBUG_INFO ("DL Message for " MACSTR ". Type 0x%02X", MAC2STR (address), msgType);
265  DEBUG_DBG ("Data: %.*s", len, data);
266 
267  if (msgType == USERDATA_GET || msgType == USERDATA_SET) {
268  const int capacity = JSON_ARRAY_SIZE (25) + 25 * JSON_OBJECT_SIZE (4);
269  DynamicJsonDocument json (capacity);
270  DeserializationError error = deserializeJson (json, data, len, DeserializationOption::NestingLimit (3));
271  if (error == DeserializationError::Ok) {
272  DEBUG_INFO ("JSON Message. Result %s", error.c_str ());
273  bufferLen = measureMsgPack (json) + 1; // Add place for \0
274  buffer = (uint8_t*)malloc (bufferLen);
275  bufferLen = serializeMsgPack (json, (char*)buffer, bufferLen);
276  encoding = MSG_PACK;
277  } else {
278  DEBUG_INFO ("Not JSON Message. Error %s", error.c_str ());
279  bufferLen++; // Add place for \0
280  buffer = (uint8_t*)malloc (bufferLen);
281  sprintf ((char*)buffer, "%.*s", len, data);
282  encoding = RAW;
283  }
284  } else {
285  bufferLen = len + 1;
286  buffer = (uint8_t*)calloc (sizeof(uint8_t),bufferLen);
287  memcpy (buffer, data, len);
288  }
289 
290

292  DEBUG_ERROR ("Error sending esp_now message to " MACSTR, MAC2STR (address));
